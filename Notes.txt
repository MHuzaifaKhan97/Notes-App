Lecture 01: Backend Setup:

- Download & Install Node.
- initialize a project => npm init --y
- Create a file server.js
- npm i express

- Test Express:

const express = require('express');
const app = express();

app.get('/',(req,res)=>{
    res.send({
        "message":"Hello World"
    });
});

app.listen('3000',()=>{
    console.log('App is running');
});
